% chapter.template.tex — pandoc LaTeX template for one PhD chapter (v5.1)
%
% Rendered from ssot.chapters.body_md (Neon SoT) by the `trios-phd render`
% subcommand. Guarantees:
%   • hero image is the first content block, width=\textwidth, [H] placement
%   • Greek glyphs (φ α λ ψ) render in italic body, captions, math
%   • long URLs and inline-code paths break across lines (no margin overflow)
%   • continuous page numbers (\setcounter{page}{$START_PAGE$})
%
% Pandoc variables consumed: body, START PAGE.

\documentclass[11pt,a4paper]{article}

% ---------- geometry ----------
\usepackage[a4paper,margin=2cm]{geometry}

% ---------- graphics + non-floating figures ----------
\usepackage{graphicx}
\usepackage{float}                    % gives us [H]

% ---------- color (must precede hyperref) ----------
\usepackage[table,dvipsnames]{xcolor}

% ---------- tables ----------
\usepackage{longtable}
\usepackage{booktabs}
\usepackage{array}
\usepackage{calc}
\usepackage{etoolbox}
\providecommand{\real}[1]{#1}

% ---------- code blocks (pandoc Shaded/Highlighting) ----------
\usepackage{fancyvrb}
\usepackage{framed}
\definecolor{shadecolor}{RGB}{248,248,248}
\newenvironment{Shaded}{\begin{snugshade}}{\end{snugshade}}
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{commandchars=\\\{\}}
% pandoc syntax-highlighting macros — render as identity (no colourisation).
\newcommand{\AlertTok}[1]{#1}\newcommand{\AnnotationTok}[1]{#1}
\newcommand{\AttributeTok}[1]{#1}\newcommand{\BaseNTok}[1]{#1}
\newcommand{\BuiltInTok}[1]{#1}\newcommand{\CharTok}[1]{#1}
\newcommand{\CommentTok}[1]{#1}\newcommand{\CommentVarTok}[1]{#1}
\newcommand{\ConstantTok}[1]{#1}\newcommand{\ControlFlowTok}[1]{#1}
\newcommand{\DataTypeTok}[1]{#1}\newcommand{\DecValTok}[1]{#1}
\newcommand{\DocumentationTok}[1]{#1}\newcommand{\ErrorTok}[1]{#1}
\newcommand{\ExtensionTok}[1]{#1}\newcommand{\FloatTok}[1]{#1}
\newcommand{\FunctionTok}[1]{#1}\newcommand{\ImportTok}[1]{#1}
\newcommand{\InformationTok}[1]{#1}\newcommand{\KeywordTok}[1]{#1}
\newcommand{\NormalTok}[1]{#1}\newcommand{\OperatorTok}[1]{#1}
\newcommand{\OtherTok}[1]{#1}\newcommand{\PreprocessorTok}[1]{#1}
\newcommand{\RegionMarkerTok}[1]{#1}\newcommand{\SpecialCharTok}[1]{#1}
\newcommand{\SpecialStringTok}[1]{#1}\newcommand{\StringTok}[1]{#1}
\newcommand{\VariableTok}[1]{#1}\newcommand{\VerbatimStringTok}[1]{#1}
\newcommand{\WarningTok}[1]{#1}

% ---------- math ----------
\usepackage{amsmath,amssymb}

% ---------- typography (DejaVu Serif: italic-Greek coverage) ----------
\usepackage{fontspec}
\setmainfont{DejaVu Serif}
\setsansfont{DejaVu Sans}
\setmonofont{DejaVu Sans Mono}
\usepackage{microtype}

% suppress section numbering — chapter slug already encodes it
\setcounter{secnumdepth}{0}

% ---------- captions (no caption package; keep main font for Greek) ----------
\usepackage{caption}
\captionsetup{labelformat=empty}

% ---------- links + URL breaking ----------
\usepackage{hyperref}
\usepackage{xurl}
\hypersetup{
  colorlinks=true,
  linkcolor={black!70!blue},
  urlcolor={black!50!blue},
  citecolor={black!60!green}
}

% ---------- pandoc plumbing ----------
\providecommand{\tightlist}{%
  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}
\providecommand{\pandocbounded}[1]{#1}

% Loosen line-breaking so paragraphs don't spill into the right margin
% when they contain long URLs / inline code that has no break opportunity.
\sloppy
\setlength{\emergencystretch}{3em}
\tolerance=2000
\hbadness=10000

% ---------- continuous page numbers ----------
% trios-phd substitutes $START_PAGE$ before invoking pandoc, so each
% chapter starts where the previous chapter ended. The cover is page 1;
% Ch.1 starts at the page after cover-final-page; Ch.2 at Ch.1-final+1; …
\setcounter{page}{$START_PAGE$}

\end{document}
