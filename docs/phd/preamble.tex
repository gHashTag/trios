% ---- R5 honesty marker — every Admitted theorem MUST be wrapped ---
%      Usage: \admittedbox{Theorem statement (Admitted in <file.v>)}
%      Also accepts an empty argument as an inline marker `\admittedbox{}`.
\newtcolorbox{admittedbox-env}{
    colback=admittedred!5,
    colframe=admittedred,
    coltitle=admittedred,
    fonttitle=\bfseries,
    title={\textsc{Admitted} (awaits Coq machine check)},
    sharp corners,
    breakable
}
\newcommand{\admittedbox}[1]{%
    \ifx\relax#1\relax
        \textsc{[Admitted]}%
    \else
        \begin{admittedbox-env}#1\end{admittedbox-env}%
    \fi
}

% ---- R4 numeric-anchor marker — every constant cites CODATA/PDG/etc.
%      Usage: \codatabox{constant = value}{source}
\newtcolorbox{codatabox-env}{
    colback=goldlight!20,
    colframe=golddark,
    sharp corners,
    breakable,
    enhanced jigsaw
}
\newcommand{\codatabox}[2]{%
    \begin{codatabox-env}\textbf{Anchor:} \(#1\) \hfill \emph{\small Source: #2}\end{codatabox-env}%
}

% ---- R14 Coq-citation marker — every \coqbox cites a .v file --------
%      Two accepted forms:
%        \coqbox{path::theorem}                                (single-arg, default Proven)
%        \coqboxFull{theorem name}{path/to/file.v}{Proven|Admitted}  (explicit form)
\newcommand{\coqbox}[1]{%
    \par\noindent\framebox[\linewidth]{%
        \parbox{0.95\linewidth}{%
            \textbf{Coq:} \texttt{#1}%
        }%
    }\par%
}
\newcommand{\coqboxFull}[3]{%
    \par\noindent\framebox[\linewidth]{%
        \parbox{0.95\linewidth}{%
            \textbf{Coq:} \texttt{#1} \ \textendash\ \texttt{#2} \ %
            (\textsc{#3})%
        }%
    }\par%
}
