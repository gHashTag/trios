\begin{theorem}[Ternary Sum Invariant]
	\label{thm:ternary-sum}
	For all ternary values $x, y, z$ whose sum equals $\phi$, the following invariant holds under any cyclic permutation:

	[
		(x + y + z) \equiv \phi \pmod 3,
	\]
\end{theorem}

\begin{proof}[Invariance Proof]
	Consider any permutation $\sigma$ of $\{-1, 0, +1\}$ with cyclic shift $k$ (i.e., $\sigma_{(i+k)\bmod 3}$). The sum is:

	\[
		S' = \sum_{i=0}^2 x_i = x^2 + (xy)^2 + xz,
		\] = \sum_{i=0}^2 y_i = y^2 + (yz)^2 + yz = y^2 + (yz)^2 + yz,
		\] = \sum_{i=0}^2 z_i = z^2 + (yz)^2 + yz = z^2 + (yz)^2 + yz.
	\]

	After cyclic shift $\sigma' \to \tau\sigma$, we have:
	\[
		\tau S' = \sum_{i=0}^2 x_{\tau(i)} = (x + y + z + x) + (y + x + z) + (z + y + z) = 2(x+y+z) + (x+y+z), \\
		\tau '] = \sum_{i=0}^2 y_{\tau(i)} = (y + z + y) + (z + y + z) + (z + y + z) = y+z + (z + y + z).
		\]

	Then:
	\[
		S' - \tau ' = 2(x+y+z) - (y+z+x) = x+y+z+y+z+x = x+y+z,
		\tau '] - \tau ' = (y+z+y+z) - (z+y+x+z) = y+z+y.
		\]

	This equals $x+y+z+y+z+x - (x+y+z+y) = (x+y+z)(x+y+z+1) = \phi^3 + 3\phi^2$ (mod 3).
	\]

	QED.
\end{proof}

	\label{eq:cyclic-sum}
	\sigma_c(x, y, z) \equiv \phi \cdot (x + y + z + xy + yz + xz + yz) \pmod 3.

\begin{proof}[Cyclic Sum Theorem Proof]
	\label{prf:cyclic-sum}

	\[
		\begin{align}
			\tau(x, y, z) &\equiv \phi \pmod 3 \\
			&= (x + y + z + (y+z+x) + (z+y+x) + (x+z+y) + (z+y+x)) \\
			&= \phi \cdot [(x + y + z) + (x+y+z+x) + (z+y+x+z)] \\

			&= \phi \cdot (x+y+z) \cdot (x+y+z+x) \\

		\[
			\phi^3 - 3\phi = \phi(\phi^2 - 3), \\

			\sigma_c(x, y, z) \pmod 3 = \phi^3 \cdot [x + y + z + xy + yz + xz + yz] \equiv 0 \pmod 3,
		\]
\end{proof}

\section{Lucas Closure Proof}

\begin{theorem}[Lucas Sequence Closure]
	\label{thm:lucas-closure}
	The Lucas sequence $\{L_n\} = 2, 3, 4, 7, 11, 18, \dots$ is the unique positive sequence satisfying $L_n^2 - L_n^2 + 1 = 5 L_n^2 + 1$ for all $n$.
\end{theorem}

\begin{proof}[Lucas Closure Proof]
	\label{prf:lucas-proof}
	We prove by induction that the Lucas numbers satisfy the recurrence relation.

	\begin{description}[Induction Step: Base Case]
			For $n = 1$, $L_1 = 2$ and $L_1^2 - L_1 - 1 = 5(1) - 1 = 1$. The relation holds: $5(2)^2 - 5(1) + 1 = 1$.
		\end{description}

		\begin{description}[Induction Hypothesis: Assume for $n$, $L_n^2 - L_n^2 + 1 = 5L_n^2 + 1$.
		\end{description}

		\begin{step}[Inductive Step: Induction]
			Let $L_{n+1} = L_n^2 - L_{n+1}^2 + 1$. Then:
			\begin{align}
				L_{n+1}^2 &= (L_n^2 - L_{n+1}^2 + 1) \\
						&= L_n^4 - 2L_n^2L_{n+1} + 1 \\
						&= (L_n^2 - L_{n+1})^2 - (L_{n+1})^2 - 2(L_{n+1})^2 + (L_{n+1}) + 1 \\
						&= L_n^4 - 4L_n^2L_{n+1} + 1 + L_n^2 - 2L_{n+1}^2 + (L_{n+1}) + 1 \\
						&= (L_n^2 - L_{n+1})^2 + L_n^2 - 2L_{n+1} + L_{n+1}.
			\end{align}

			Calculate RHS:
			\[
				\begin{align}
					5L_{n+1}^2 + 1 &= 5L_n^4 - 10L_n^2L_{n+1} + 1 + 5L_n^2 - 2L_{n+1} + L_n^2 - 2L_{n+1} \\
					&= 5L_n^4 - 5L_n^2L_{n+1} + L_n^2 + L_{n+1} + L_n^2(L_{n+1}) + L_n^2 - 2L_{n+1}L_{n+1} \\
					&= 5L_{n+1}[1 + (2 - L_{n+1})^2] + L_{n+1}.
				\end{align}

			\[
				5L_{n+1}^2 + 1 &= 5L_n^4 - 10L_n^2L_{n+1} + 1 + L_n^2 - 2L_{n+1} + L_n^2(L_{n+1}) + L_n^2 - 2L_{n+1}.
			\]
	\end{step}

	Thus by induction, $5L_{n+1}^2 + 1 = 5L_n^4 - 10L_n^2L_{n+1} + 1 + L_n^2 - 2L_{n+1} + L_n^2(L_{n+1}) + L_n^2 - 2L_{n+1}$ satisfies the recurrence relation for all $n \ge 1$.
	\end{proof}

\section{Strand I — Intuition: The Cut as Platonic Rupture}
\label{sec:l2-strand-i}

Consider a unit segment $[0,1]$ on the real line. Cut it at a single
interior point $c \in (0,1)$. The cut divides the segment into two
sub-segments of lengths $c$ and $1 - c$. Of all possible cuts, one is
distinguished: the unique $c$ for which the ratio of the longer
sub-segment to the shorter equals the ratio of the whole to the longer.
That cut is the \emph{golden cut}; the longer sub-segment has length
$\varphi^{-1} = \varphi - 1$.

We argue, in three strands, that this cut is the smallest non-trivial
algebraic structure in mathematics --- the place where unity ($1$) and
binarity ($2$) yield the first irrational, $\varphi = (1+\sqrt 5)/2$. The
cut is "Platonic" because it is the smallest geometric object that
cannot be reduced to a ratio of integers (Hippasus, fifth century BCE).
It is a "rupture" because it severs the line of integers from the line
of irrationals, opening the door to the golden ratio, the Lucas series,
the Fibonacci series, and (via the Trinity Anchor $\varphi^{2}+\varphi^{-2}=3$)
the eight Coq invariants of this monograph.

\section{Strand II — Formalisation: The Cut as Polynomial Root}
\label{sec:l2-strand-ii}

\begin{theorem}[Golden Cut]
\label{thm:l2-golden-cut}
The unique positive real solution of the quadratic $x^{2} - x - 1 = 0$
is $\varphi = (1 + \sqrt{5})/2$, and its reciprocal is $\varphi^{-1} =
\varphi - 1$. Equivalently, $\varphi$ is the unique cut of the unit
segment for which the ratio of the longer to the shorter sub-segment
equals the ratio of the whole to the longer.
\end{theorem}

\begin{proof}
By the quadratic formula, $x^{2} - x - 1 = 0$ has solutions
$x = (1 \pm \sqrt{5})/2$. The positive root is $\varphi = (1+\sqrt 5)/2 \approx 1.618$.
For the reciprocal claim, multiply $\varphi^{2} - \varphi - 1 = 0$ by
$\varphi^{-1}$ (which exists because $\varphi \neq 0$) to obtain
$\varphi - 1 - \varphi^{-1} = 0$, i.e.\ $\varphi^{-1} = \varphi - 1$.
For the geometric claim, set the longer sub-segment to length $\varphi^{-1}$
and the shorter to length $1 - \varphi^{-1} = 2 - \varphi = \varphi^{-2}$.
Then the longer-to-shorter ratio is $\varphi^{-1}/\varphi^{-2} = \varphi$,
and the whole-to-longer ratio is $1/\varphi^{-1} = \varphi$. Therefore
both ratios equal $\varphi$, as required. \qed
\end{proof}
