\begin{equation}

\end{equation}

\begin{equation}

\end{definition}

\begin{definition}[Electroweak Gauge Group]
\label{def:su2}

\begin{equation}

\end{equation}
\end{definition}

\begin{theorem}[Pauli Matrices]
\label{thm:pauli}

\begin{equation}
\sigma^x = \begin{pmatrix}0 & 1 \\ 1 & 0\end{pmatrix}, \quad
\sigma^y = \begin{pmatrix}0 & -i \\ i & 0\end{pmatrix}, \quad

\end{equation}

\begin{equation}
[\sigma^i, \sigma^j] = 2i\epsilon^{ijk} \sigma^k
\end{equation}
\end{theorem}
